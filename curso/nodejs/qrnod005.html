G10 - Exercício final

Baseado na tabela de Curso faça:

  • Criar APIs para listar, inserir, atualizar e deletar cursos.
  • Criar testes para as APIs criadas. 
CREATE TABLE Curso (
    ID int NOT NULL AUTO_INCREMENT,
    Nome varchar(255) DEFAULT NULL,
    Link varchar(255) DEFAULT NULL,
    Campus varchar(255) DEFAULT NULL,
    PRIMARY KEY (ID)
);
CREATE TABLE Usuario (
        Id INT AUTO_INCREMENT PRIMARY KEY,
        Nome VARCHAR(200) NOT NULL,
        Email VARCHAR(250) UNIQUE NOT NULL,
        CPF CHAR(11) NULL,
        DDD SMALLINT NULL,
        Telefone NUMERIC(9) NULL,
        DataNiver DATE NULL,
        Obs TEXT null
);
CREATE TABLE IF NOT EXISTS Produtos (
        id INT AUTO_INCREMENT PRIMARY KEY, -- ID único para cada produto, auto-incrementa a cada novo registro
        nome VARCHAR(255) NOT NULL,        -- Nome do produto, não pode ser nulo
        preco DECIMAL(10, 2) NOT NULL,     -- Preço do produto, com 2 casas decimais e não pode ser nulo
        estoque INT DEFAULT 0,             -- Quantidade em estoque (opcional, com valor padrão 0)
        data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP -- Data e hora de criação, preenchida automaticamente
);
